cmake_minimum_required(VERSION 3.10)
project(bls12_381_python_bindings)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find Python and pybind11
find_package(Python3 COMPONENTS Interpreter Development REQUIRED)
find_package(pybind11 REQUIRED)

# Include directories for MCL and BLS12-381
include_directories(3rd/hyrax-bls12-381/3rd/mcl/include)
include_directories(3rd/hyrax-bls12-381/src)

# Add the MCL library (assuming it's already built)
# You may need to adjust this path based on your MCL installation
link_directories(3rd/hyrax-bls12-381/3rd/mcl/lib)

# Create the Python module
pybind11_add_module(bls12_381_python bls12_381_wrapper.cpp)

# Link against MCL libraries
target_link_libraries(bls12_381_python mcl)

# Set compiler flags
if(MSVC)
    target_compile_options(bls12_381_python PRIVATE /std:c++17)
else()
    target_compile_options(bls12_381_python PRIVATE -std=c++17)
endif()
